**快速排序就是个二叉树的前序遍历，归并排序就是个二叉树的后序遍历**
### 搜索二叉树
```Csharp
class Solution {
    // 递归，利用二叉搜索树特点
    public TreeNode searchBST(TreeNode root, int val) {
        if (root == null || root.val == val) {
            return root;
        }
        // 如果小于根节点
        if (val < root.val) {
            return searchBST(root.left, val);
        } else {
        // 反之
            return searchBST(root.right, val);
        }
    }
}
```

### 是不是有效搜索树
```Csharp
class Solution {
    public long pre = Long.MIN_VALUE;
    public boolean isValidBST(TreeNode root) {
        if (root == null) {
            return true;
        }
        if (!isValidBST(root.left)){
            return false;
        }
        if (root.val <= pre) {
            return false;
        }
        pre = root.val;
        return isValidBST(root.right);
    }
}
```

```java
boolean isValidBSTRecursive(TreeNode root, TreeNode min,TreeNode max){
	if(root == null)
		return true;
	if(min != null && root.val <= min.val) return false;
	if(max != null && root.val >= max.val) return false;
	return isValidBSTRecursive(root.left,min,root) && isValidBSTRecursive(root.right,root,max)
}
```

### 在BST中插入一个元素

```java
TreeNode Insert(TreeNode root,int val){
	if(root == null){
		return new TreeNode(val);
	}
	if(root.val < val){
		root.right = Insert(root.right,val)
	}
	if(root.val > val){
		root.left = Insert(root.left,val)
	}
	return root;
}
```

通过中序遍历和后序遍历构建二叉树

```javascript
var buildTree = function(inorder, postorder) {
    let post_idx;
    const idx_map = new Map();
    const helper = (in_left, in_right) => {
        // 如果这里没有节点构造二叉树了，就结束
        if (in_left > in_right) {
            return null;
        }

        // 选择 post_idx 位置的元素作为当前子树根节点
        const root_val = postorder[post_idx];
        const root = new TreeNode(root_val);

        // 根据 root 所在位置分成左右两棵子树
        const index = idx_map.get(root_val);

        // 下标减一
        post_idx--;
        // 构造右子树
        root.right = helper(index + 1, in_right);
        // 构造左子树
        root.left = helper(in_left, index - 1);
        return root;
    }

    // 从后序遍历的最后一个元素开始
    post_idx = postorder.length - 1;

    // 建立（元素，下标）键值对的哈希表
    let idx = 0;
    inorder.forEach((val, idx) => {
        idx_map.set(val, idx);
    });
    return helper(0, inorder.length - 1);
};

```

从前序和后序构建二叉树
![[Pasted image 20240228192235.png]]
```java
class Solution {
    int[] pre, post;
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        this.pre = pre;
        this.post = post;
        return make(0, 0, pre.length);
    }

    public TreeNode make(int i0, int i1, int N) { 
        if (N == 0) return null;
        TreeNode root = new TreeNode(pre[i0]);
        if (N == 1) return root;

        int L = 1;
        for (; L < N; ++L)
            if (post[i1 + L-1] == pre[i0 + 1])
                break;

        root.left = make(i0+1, i1, L);
        root.right = make(i0+L+1, i1+L, N-1-L);
        return root;
    }
}
```

### 判断对称二叉树

```java
public bool symmetricTree(TreeNode root){
	check(root,root);
}
bool check(TreeNode p,TreeNode q){
	if(p == null && q == null){
		return true
	}
	if(p == null || q == null){
		return false
	}
	return check(p.left,q.right) && check(p.right,q.left) && p.val == q.val;
}
```

### 二叉树深度

```java
//回溯 算法
int maxDepth = 0;
int res = 0;
public int maxDepth(TreeNode root){
	maxDepth++;
	if(root.left == null && root.right == null){
		res = Math.max(res,maxDepth);
	}
	maxDepth(root.left);
	maxDepth(root.right);
	maxDepth--;
}
//递归算法 
public int maxDepth(TreeNode root){
	if(root == null){
		return 0
	}
	int left  = maxDepth(root.left);
	int right = maxDepth(root.right);
	int maxDepth = Math.max(left,right)+1;
	return maxDepth;
}
```

### 二叉树直径

```java
// N的平方复杂度
int maxDiameter = 0;
public int diameterOfBinaryTree(TreeNode root){
	traverse(root);
	return maxDiameter;
}
traverse(TreeNode root){
	if(root == null){
		return;
	}
	int left = maxDepth(root.left)
	int right = maxDepth(root.right)
	maxDiameter = Math.max(maxDiameter,left+right)
	traverse(root.left);
	traverse(root.right);
}
```
