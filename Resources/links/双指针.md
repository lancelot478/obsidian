#Algorithm

```Java
class Solution {
    int all_area = 0;
    int area = 0;
    public int trap(int[] height) {
        for (int i = 0; i < height.length; i++) {
            int v = height[i];
            area += v;
        }
        int l = 0;
        int r = height.length - 1;
        int k = 1;
        while (l <= r){
            while (l <= r && height[l] < k) l++;
            while (l <= r && height[r] < k) r--;
            all_area += r - l + 1;
            k++;
        }
        return all_area - area;
    }
}
```

```Cpp
class Solution {
    //其实右边有足够高的，左边底的绝对都能放进去水
public:
    int trap(vector<int>& height) 
    {
        int n = height.size();
        int l = 1,r = n-2;
        int leftmax = height[0],rightmax = height[n-1];
        int ret = 0;
        while(l<=r)
        {
            while(leftmax <= rightmax&&l<=r)
            {
                if(height[l] < leftmax)
                    ret += leftmax - height[l];
                else
                    leftmax = height[l];
                l++;
            }
            
            while(leftmax > rightmax&&l<=r)
            {
                if(height[r] < rightmax)
                    ret += rightmax - height[r];
                else
                    rightmax = height[r];
                r--;
            }
        }
        return ret;
    }
};
```
