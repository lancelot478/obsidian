private int num = 0;  
public int count(int[] a, int n)  
{  
    num = 0;  
    mergeSortCounting(a, 0, n - 1);  
      
    return num;  
}  
  
private void mergeSortCounting(int[] a, int left, int right)  
{  
    if (left >= right) return;  
      
    int mid = (left + right) / 2;  
    mergeSortCounting(a, left, mid);  
    mergeSortCounting(a, mid+1, right);  
    merge(a, left, mid, right);  
}  
  
private void merge(int[] a, int left, int mid, int right)  
{  
    int i = left, j = mid + 1;  
    int k = 0;  
    int[] temp = new int[right - left +1];  
    while (i <= mid && j <= right)  
    {  
        if (a[i] <= a[j])  
        {  
            temp[k++] = a[i++];  
        }  
        else  
        {  
            num += mid - i + 1;  
            temp[k++] = a[j++];  
        }  
    }  
    while (i <= mid)  
    {  
        temp[k++] = a[i++];  
    }  
  
    while (j <= right)  
    {  
        temp[k++] = a[j++];  
    }  
  
    for (int m = 0; m < temp.Length; m++)  
    {  
        a[left + m] = temp[m];  
    }  
}

public static int[,] initializeMatrix(int n)  
{  
    var result = new int[n, n];  
      
    for (int i = 0; i < n; i++)  
    {  
        for (int j = 0; j < n; j++)  
        {  
            result[i][j] = 1;  
        }  
    }    return result;  
}  
  
public static int[,] BruteForce(int[,] p,int[,] q,int n)  
{  
    var result = new int[n, n];  
    for (int i = 0; i < n; i++)  
    {  
        for (int j = 0; j < n; j++)  
        {  
            for (int k = 0; k < n; k++)  
            {  
                result[n,n] += p[i,k] * q[k,j];  
            }  
        }    }  
    return result;  
}  
  
public static int[,] DivideAndConquer(int[,] p, int[,] q, int n)  
{  
    var result = new int[n, n];  
    if(n == 2){  
        result = BruteForce(p,q,n);              
        return result;  
    }  
  
    int m = n / 2;  
    int[,] p1 = QuarterMatrix(p, 0, m - 1, 0, m - 1);  
    int[,] p2 = QuarterMatrix(p, m, n, 0, m - 1);  
    int[,] p3 = QuarterMatrix(p, 0, m - 1, m, n);  
    int[,] p4 = QuarterMatrix(p, m, n, m, n);  
      
    int[,] q1 = QuarterMatrix(q, 0, m - 1, 0, m - 1);  
    int[,] q2 = QuarterMatrix(q, m, n, 0, m - 1);  
    int[,] q3 = QuarterMatrix(q, 0, m - 1, m, n);  
    int[,] q4 = QuarterMatrix(q, m, n, m, n);  
  
    var result1 = QuarterMatrix(result, 0, m - 1, 0, m - 1);  
    var result2 = QuarterMatrix(result, m, n,  0, m - 1);  
    var result3 = QuarterMatrix(result, 0, m - 1, m, n);  
    var result4 = QuarterMatrix(result, m, n,  m, n);  
  
    result1 = AddMatrix(DivideAndConquer(p1,q1,m),DivideAndConquer(p2,q3,m),m);  
    result2 = AddMatrix(DivideAndConquer(p3,q1,m),DivideAndConquer(p4,q3,m),m);  
    result3 = AddMatrix(DivideAndConquer(p1,q2,m),DivideAndConquer(p2,q4,m),m);  
    result4 = AddMatrix(DivideAndConquer(p3,q2,m),DivideAndConquer(p4,q4,m),m);  
      
    result = TogetherMatrix(result1,result2,result3,result4,m);  
      
    return result;  
}  
  
public static int[,] TogetherMatrix(int[,] p1, int[,] p2, int[,] p3, int[,] p4, int n)  
{  
    var result = new int[2 * n, 2 * n];  
  
    for (int i = 0; i < n; i++)  
    {  
        for (int j = 0; j < n; j++)  
        {  
            result[i, j] = p1[i, j];  
            result[i+ n, j ] = p2[i, j];  
            result[i, j + n] = p3[i, j];  
            result[i + n, j + n] = p4[i, j];  
        }  
    }    return result;  
}  
public static int[,] AddMatrix(int[,] p, int[,] q, int n)  
{  
    var result = new int[n, n];  
    for (int i = 0; i < n; i++)  
    {  
        for (int j = 0; j < n; j++)  
        {  
            result[i,j] = p[i,j] + q[i,j];  
        }  
    }    return result;  
}  
public static int[,] QuarterMatrix(int[,] p, int rowStart, int rowEnd, int colStart, int colEnd)  
{  
    var length = rowEnd - rowStart + 1;  
    var temp = new int[length,length];  
    for (int i = rowStart; i <= rowEnd; i++)  
    {  
        for (int j = colStart; j <= colEnd; j++)  
        {  
            temp[i-rowStart,j-colStart] = p[i,j];  
        }  
    }  
    return temp;  
}
