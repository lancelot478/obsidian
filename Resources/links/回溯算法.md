### 全排列

```go
// permute 函数用于生成数组的全排列  
func permute(nums []int) [][]int {  
    var res [][]int  
    // 调用回溯函数  
    backtrack(nums, 0, &res)  
    return res  
}  
  
// backtrack 是回溯核心函数  
func backtrack(nums []int, start int, res *[][]int) {  
    // 当 start 达到数组末尾，说明找到一个排列  
    if start == len(nums) {  
       // 复制当前排列，防止修改原数组  
       tmp := make([]int, len(nums))  
       copy(tmp, nums)  
       *res = append(*res, tmp)  
       return  
    }  
  
    // 遍历数组，从 start 开始交换位置  
    for i := start; i < len(nums); i++ {  
       // 交换元素  
       nums[start], nums[i] = nums[i], nums[start]  
       // 继续递归下一个位置  
       backtrack(nums, start+1, res)  
       // 撤销交换，回溯到上一状态  
       nums[start], nums[i] = nums[i], nums[start]  
    }
}
```

var solvedNQueue = function(n){
	var res = [];
	var board = Array.from({length:n},()=> Array(n).fill('.'))

	var backTrack = function(board,row){
		if(row == board.length){
			res.push(JSON.parse(JSON.stringify(board))); //deep-copy
		}
		var n = board[row].length;
		for(int col = 0; col<n;col++){
			if(!IsValid(board,row,col)){
				continue;
			}
			board[row][col] = "Q";
			backTrack(board,row+1);
			board[row][col] = ".";
		}
	}
	var isValid = function(board, row, col) {
        var n = board.length;
        // 检查列是否有皇后互相冲突
        for (var i = 0; i < n; i++) {
            if (board[i][col] === 'Q') {
                return false;
            }
        }
        // 检查右上方是否有皇后互相冲突
        for (var i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {
            if (board[i][j] === 'Q') {
                return false;
            }
        }
        // 检查左上方是否有皇后互相冲突
        for (var i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--) {
            if (board[i][j] === 'Q') {
                return false;
            }
        }
        return true;
    };

	backTrack(board,0);

	return res;
}
